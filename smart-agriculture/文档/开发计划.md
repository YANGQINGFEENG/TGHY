#### 技术实现
```typescript
// 硬件端上传执行器状态（支持数字格式）
POST /api/actuators/WP-002
{
  "state": 0,    // 0=off, 1=on
  "mode": 1      // 0=auto, 1=manual
}

// 前端实时更新（每2秒刷新）
const fetchActuators = useCallback(async () => {
  const response = await fetch('/api/actuators', {
    cache: 'no-store',
    headers: { 'Cache-Control': 'no-cache' }
  })
  const result = await response.json()
  
  // 只显示1小时内有数据的执行器
  const validActuators = result.data.filter((a: Actuator) => {
    if (!a.last_update) return false
    const hoursDiff = (Date.now() - new Date(a.last_update).getTime()) / (1000 * 60 * 60)
    return hoursDiff < 1
  })
  
  setActuators(validActuators)
}, [])

// 状态变化立即更新UI
if (result.success) {
  setActuators(prev => 
    prev.map(a => 
      a.id === actuatorId 
        ? { ...a, state: newState, last_update: new Date().toISOString() }
        : a
    )
  )
  setTimeout(() => fetchActuators(), 500)
}
```

results.forEach(({ sensorId, data }) => {

  - 实时显示传感器状态（正常/延迟/异常/离线）
  - 状态判断逻辑：
    - 正常：最后更新时间 < 5分钟
    - 延迟：最后更新时间 5-30分钟
    - 异常：最后更新时间 > 30分钟
    - 离线：设备状态为 offline

#### 技术实现
```typescript
// 状态判断逻辑
const now = new Date()
const lastUpdate = sensor.last_update ? new Date(sensor.last_update) : null
const timeDiff = lastUpdate ? Math.floor((now.getTime() - lastUpdate.getTime()) / 1000 / 60) : 999

let status = '离线'
if (sensor.status === 'online') {
  if (timeDiff < 5) {
    status = '正常'
  } else if (timeDiff < 30) {
    status = '延迟'
  } else {
    status = '异常'
  }
}
```

### 4. 完整性改进

#### 完成内容
- ✅ **新增传感器类型展示**

  **土壤温度传感器（soil_temperature）**
  - 传感器ID：T-002
  - 单位：°C
  - 正常范围：15°C - 25°C
  - 预警阈值：10°C - 30°C
  - 描述：监测土壤温度，影响作物根系生长和养分吸收

  **土壤电导率传感器（ec）**
  - 传感器ID：E-001
  - 单位：μS/cm
  - 正常范围：200 - 800 μS/cm
  - 预警阈值：100 - 1500 μS/cm
  - 描述：监测土壤电导率，反映土壤盐分含量和肥力状况

  **土壤pH值传感器（ph）**
  - 传感器ID：P-001
  - 单位：pH
  - 正常范围：6.0 - 7.5 pH
  - 预警阈值：5.5 - 8.0 pH
  - 描述：监测土壤酸碱度，影响作物对养分的吸收效率
